CREATE TABLE tbl_usuario (
id_usuario int AUTO_INCREMENT PRIMARY KEY,
cd_CPF CHAR(21) NOT NULL,
nm_usuario VARCHAR(30) NOT NULL,
ds_email VARCHAR(50) NOT NULL,
dt_nascimento DATE,
tp_sexo char(1),
nm_login VARCHAR(20),
cd_senha varchar(50),
ic_ativo char(1),
tp_usuario char(1),
dt_cadastro DATE,
dt_exclusao DATE,
dt_ultimoAcesso DATE
);
